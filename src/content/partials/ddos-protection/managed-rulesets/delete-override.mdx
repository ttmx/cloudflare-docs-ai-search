import { Tabs, TabItem, Steps, DashButton } from "~/components";

### Delete a DDoS override

<Tabs syncKey="dashNewNav">
    <TabItem label="New dashboard" icon="rocket">
        <Steps>
        1. In the Cloudflare dashboard, go to the **{props.newDashPage}** page.

            <DashButton url={props.dashButtonUrl} />
        2. Go to the **{props.newDashTab}** tab.
        3. Select the override.
        4. Select **Delete deployment**.
        </Steps>
    </TabItem>
    <TabItem label="Old dashboard">
        <Steps>
            1. Log in to the [Cloudflare dashboard](https://dash.cloudflare.com/), and {props.selectAccountOrDomain}.
            2. Go to **{props.oldDashNav}**.
            3. Next to the DDoS override you wish to delete, select **Delete**.
            4. Select **Delete** to confirm the operation.
        </Steps>
    </TabItem>
</Tabs>
